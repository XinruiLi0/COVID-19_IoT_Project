For Account initialize:
	User Register: (Provides the register server to user, guard, and doctor)
		type: post
		url: http://ivmsp.us-east-1.elasticbeanstalk.com/Login/userLogin
		Content-Type: application/json
		body(submit data): userName, userEmail, userPassword, userRole
			example:
			{
				"userName": "a"
				"userEmail": "b", 
				"userPassword": "c", 
				"userRole": 1
			}
		return data:
			example:
			{
				"result":"success", 
				"message": "Success."
			}

	Guard Register: (Provides the register server to guard)
		type: post
		url: http://ivmsp.us-east-1.elasticbeanstalk.com/Login/userLogin
		Content-Type: application/json
		body(submit data): userName, userEmail, userPassword, address, latitude, longitude
			example:
			{
				"userName": "a"
				"userEmail": "b", 
				"userPassword": "c", 
				"address": "d",
				latitude: -1.1234567,
				longitude: 1.1234567
			}
		return data:
			example:
			{
				"result":"success", 
				"message": "Success."
			}
	
	User Login: (Provides login to user, guard, and doctor)

		body(submit data): 
			example:

				"userRole": 1

		return data:

For User App & Web:
	User self-check: (Provides self-check server to user app and web)
		type: post
		url: http://ivmsp.us-east-1.elasticbeanstalk.com/Home/checkUserStatus
		Content-Type: application/json
		body(submit data): 
			example:
			{
				"userEmail": "a", 
				"userPassword": "b"
			}
		return data:
			example:

			{
				0: 	{
						"UserName": "c", 
						"UserStatus": 0
					},
				1:	{	// if the user has contact history with confirmed patient
						"Address": "d",
						"StartTime": "e",
						"EndTime": "f"
					}

			}

For Doctor Web:
	Patient status check: (Provides health check server to doctor)
		type: post
		url: http://ivmsp.us-east-1.elasticbeanstalk.com/Home/checkPatientStatus
		Content-Type: application/json
		body(submit data): userEmail, userPassword, visitorEmail
			example:
			{
				"userEmail": "a",
				"userPassword": "b",
				"visitorEmail": "c"
			}
		return data:
			example:
			{
				"UserName": "d",
				"UserEmail": "c",
				"UserStatus": 0
			}
	
	Patient status update: (Provides health update server to doctor)
		type: post
		url: http://ivmsp.us-east-1.elasticbeanstalk.com/Home/updatePatientStatus
		Content-Type: application/json
		body(submit data): userEmail, userPassword, visitorEmail, status
			example:
			{
				"userEmail": "a",
				"userPassword": "b",
				"visitorEmail": "c",
				"status": 0
			}
		return data:
			example:
			{
				"UserName": "d",
				"UserEmail": "c",
				"UserStatus": 0
			}

For Guard Web:
	Get guard devices: (Provides checking server to guard)
		type: post
		url: http://ivmsp.us-east-1.elasticbeanstalk.com/Home/getGuardDevices
		Content-Type: application/json
		body(submit data): userEmail, userPassword
			example:
			{
				"userEmail": "a",
				"userPassword": "b"
			}
		return data:
			example:
			{
				0: 	{
						"DeviceID": "c",
						"Description": "d"
					},
				1:	{
						"DeviceID": "e",
						"Description": "f"
					}
			}
	
	Register guard devices: (Provides device adding server to guard)
		type: post
		url: http://ivmsp.us-east-1.elasticbeanstalk.com/Home/registerGuardDevice
		Content-Type: application/json
		body(submit data): userEmail, userPassword, deviceID, deviceDescription
			example:
			{
				"userEmail": "a",
				"userPassword": "b",
				"deviceID": "c",
				"deviceDescription": "d"
			}
		return data:
			example:
			{
				0: 	{
						"DeviceID": "c",
						"Description": "d"
					},
				1:	{
						"DeviceID": "e",
						"Description": "f"
					}
			}
	
	Delete guard devices: (Provides device removing server to guard)
		type: post
		url: http://ivmsp.us-east-1.elasticbeanstalk.com/Home/deleteGuardDevice
		Content-Type: application/json
		body(submit data): userEmail, userPassword, deviceID
			example:
			{
				"userEmail": "a",
				"userPassword": "b",
				"deviceID": "c"
			}
		return data:
			example:
			{
				0: 	{
						"DeviceID": "c",
						"Description": "d"
					},
				1:	{
						"DeviceID": "e",
						"Description": "f"
					}
			}

	Entrance monitor: (Provides device info updating server to guard's web)
		type: post
		url: http://ivmsp.us-east-1.elasticbeanstalk.com/Home/visitorInfoCheck
		Content-Type: application/json
		body(submit data): deviceID
			example:
			{
				"deviceID": "a"
			}
		return data:
			example:
			{
				"UserName": "b",
				"UserStatus": 0,
				"VisitorTemperature": 37.1
				"LastUpdated": "1362"  // 1362 seconds after the last update
			}
